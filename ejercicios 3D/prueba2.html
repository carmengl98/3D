var g_points = []; // The array for the position of a mouse press function click(ev, gl, canvas, a_Position) { var x = ev.clientX; // x coordinate of a mouse pointer var y = ev.clientY; // y coordinate of a mouse pointer var rect = ev.target.getBoundingClientRect(); x = ((x - rect.left) - canvas.width / 2) / (canvas.width / 2); y = (canvas.height / 2 - (y - rect.top)) / (canvas.height / 2); // Store the coordinates to g_points array g_points.push(x); g_points.push(y); // Clear gl.clear(gl.COLOR_BUFFER_BIT); var len = g_points.length; for (var i = 0; i < len; i += 2) { // Pass the position of a point to a_Position variable gl.vertexAttrib3f(a_Position, g_points[i], g_points[i + 1], 0.0); // Draw gl.drawArrays(gl.POINTS, 0, 1); } }
